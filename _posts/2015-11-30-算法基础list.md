#include <stdio.h>

char * strcopy(const char * src, char * des){

	if(NULL == src || NULL == des){
		return NULL;
	}
	char *p = des;

	while(*src != '\0'){
		*des = *src;
		des++;
		src++;
	}
	*des = '\0';
	return p;
}

int main(){

	const char *src = "abc";
	char *des;
	des = strcopy(src, des);
	printf("%s\n", des);
}
